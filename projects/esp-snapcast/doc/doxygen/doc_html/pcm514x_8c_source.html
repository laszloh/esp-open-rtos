63 pcm514x_error_t pcm514x_init(pcm514x_t* dev)
64 {
65  int ret;
66 
67  pcm514x_priv_t *pcm514x;
68  pcm514x = malloc(sizeof(pcm514x_priv_t));
69  if(pcm514x == NULL)
70  return PCM514x_ERR_NOMEM;
71  memset(pcm514x, 0, sizeof(pcm514x_priv_t));
72  dev->io = pcm514x;
73 
74  // setup the communication interface
75  switch(dev->mode) {
76 
77  case HWCFG_SPI:
78  // set up the interface
79  gpio_enable(dev->spi.cs, GPIO_OUTPUT);
80  gpio_write(dev->spi.cs, true);
81  spi_init(SPI_BUS, SPI_MODE0, SPI_FREQ_DIV_8M, true, SPI_LITTLE_ENDIAN, true);
82 
83  dev->io->write = spi_write;
84  dev->io->read = spi_read;
85  dev->io->write_block = spi_write_block;
86  dev->io->read_block = spi_read_block;
87  break;
88 
89  case HWCFG_HW:
90  debug(3, "Hardwired mode is not fully supported!");
91  return PCM514x_ERR_OK;
92 
93  case HWCFG_I2C:
94  default:
95  // we currently only support the SPI mode
96  debug(1, "I2C is currently not supported")
97  return PCM514x_ERR_HWCFG_UNSUPPORTED;
98  }
99 
100  // reset the PCM514x
101  ret = write_register(dev, PCM512x_RESET, (PCM512x_RSTR | PCM512x_RSTM),
102  (PCM512x_RSTR | PCM512x_RSTM));
103  if(ret != PCM514x_ERR_OK)
104  return ret;
105 
106  // setup the clk sources
107  if(dev->clk.out_en) {
108  // clk out from the esp is currently not supported
109  debug(1, "SCK generation through PLL and external clock is not supported!");
110  return PCM514x_ERR_HWCFG_UNSUPPORTED;
111 
112  // we are supplying the pcm5412 with our clock
113 // gpio_set_iomux_function(4, IOMUX_GPIO4_FUNC_CLOCK_XTAL);
114 // ret = pcm514x_pll_setup(dev);
115 // if(ret != PCM514x_OK)
116 // return ret;
117  } else {
118  // setup BCLK as clk source
119 
120  // ignore clk error
121  ret = write_register(dev, PCM512x_ERROR_DETECT,
122  PCM512x_IDCH, PCM512x_IDCH);
123  if(ret != PCM514x_ERR_OK) {
124  debug(1, "Failed to set ignore bits in register PCM512x_ERROR_DETECT: %d", ret);
125  return ret;
126  }
127 
128  // set the PLL to the BCLK
129  ret = write_register(dev, PCM512x_PLL_REF,
130  PCM512x_SREF, PCM512x_SREF);
131  if(ret != PCM514x_ERR_OK) {
132  debug(1, "Failed to switch PLL to BCLK: %d", ret);
133  return ret;
134  }
135  }
136 
137  // default to standby mode
138  ret = write_register(dev, PCM512x_POWER,
139  PCM512x_RQST, PCM512x_RQST);
140  if(ret != PCM514x_ERR_OK) {
141  debug(1, "Failed to request standby: %d", ret);
142  return ret;
143  }
144 
145  return PCM514x_ERR_OK;
146 }

148 pcm514x_error_t pcm514x_init_dai(pcm514x_t* dev, pcm514x_dai_t *dai)
149 {
150  uint8_t alen;
151  uint8_t clock_config;
152  uint8_t master_mode;
153  int ret;
154 
155  switch(dai->width){
156  case 16:
157  alen = PCM512x_ALEN_16;
158  break;
159  case 20:
160  alen = PCM512x_ALEN_20;
161  break;
162  case 24:
163  alen = PCM512x_ALEN_24;
164  break;
165  case 32:
166  alen = PCM512x_ALEN_32;
167  break;
168  default:
169  debug(1, "Unsupported sample width %d supplied", dai->width);
170  return PCM514x_ERR_PCM_UNSUPPORTED;
171  }
172  ret = write_register(dev, PCM512x_I2S_1, PCM512x_ALEN, alen);
173  if(ret != PCM514x_ERR_OK){
174  debug(1, "Failed to set pcm sample width: %d", ret);
175  return ret;
176  }
177 
178  switch(dai->mode){
179  case DAI_CBS_CFS:
180  clock_config = 0;
181  master_mode = 0;
182 
183  // enable clock divider autoset
184  ret = write_register(dev, PCM512x_ERROR_DETECT, PCM512x_DCAS, 0);
185  if(ret != PCM514x_ERR_OK) {
186  debug(1, "Failed to enable clock divider autoset: %d", ret);
187  return ret;
188  }
189  break;
190  case DAI_CBM_CFM:
191 // clock_config = PCM512x_BCKO | PCM512x_LRKO;
192 // master_mode = PCM512x_RLRK | PCM512x_RBCK;
193  debug(2, "BCLK & LRCLK master nor supported by ESP8266!");
194  case DAI_CBM_CFS:
195 // clock_config = PCM512x_BCKO;
196 // master_mode = PCM512x_RBCK;
197  debug(1, "PCM as I2S not implemented");
198  default:
199  return PCM514x_ERR_PCM_UNSUPPORTED;
200  }
201  ret = write_register(dev, PCM512x_BCLK_LRCLK_CFG,
202  PCM512x_BCKP | PCM512x_BCKO | PCM512x_LRKO, clock_config);
203  if(ret != PCM514x_ERR_OK) {
204  debug(1, "Failed to set i2s clock mode: %d", ret);
205  return ret;
206  }
207  ret = write_register(dev, PCM512x_MASTER_MODE,
208  PCM512x_RLRK | PCM512x_RBCK, master_mode);
209  if(ret != PCM514x_ERR_OK) {
210  debug(1, "Failed to set i2s master mode: %d", ret);
211  return ret;
212  }
213 
214  // (re)synchronize DAI clocks
215  ret = write_register(dev, PCM512x_SYNCHRONIZE,
216  PCM512x_RQSY, PCM512x_RQSY_HALT);
217  if(ret != PCM514x_ERR_OK) {
218  debug(1, "Failed to halt clocks: %d", ret);
219  return ret;
220  }
221  ret = write_register(dev, PCM512x_SYNCHRONIZE,
222  PCM512x_RQSY, PCM512x_RQSY_RESUME);
223  if(ret != PCM514x_ERR_OK) {
224  debug(1, "Failed to resume clocks: %d", ret);
225  return ret;
226  }
227 
228  return PCM514x_ERR_OK;
229 }

291 pcm514x_error_t pcm514x_set_audio_ramp(pcm514x_t* dev, const pcm514x_audio_ramp_t* ramp)
292 {
293  uint8_t val;
294  int ret;
295 
296  // configure volume ramp up and down
297  val = (ramp->vol_ramp_down.freq << PCM512x_VNDF_SHIFT) | (ramp->vol_ramp_down.step << PCM512x_VNDS_SHIFT);
298  val |= (ramp->vol_ramp_up.freq << PCM512x_VNUF_SHIFT) | (ramp->vol_ramp_up.step << PCM512x_VNUS_SHIFT);
299  ret = write_register(dev, PCM512x_DIGITAL_MUTE_1,
300  (PCM512x_VNDF | PCM512x_VNDS | PCM512x_VNUF | PCM512x_VNUS), val);
301  if(ret != PCM514x_ERR_OK) {
302  debug(1, "Failed to write register 0x%02x to 0x%02x: %d", PCM512x_DIGITAL_MUTE_1, val, ret);
303  return ret;
304  }
305  // configure emergency ramp down (f.e. on SCKL loose)
306  val = (ramp->emergency_down.freq << PCM512x_VEDF_SHIFT) | (ramp->emergency_down.step << PCM512x_VEDS_SHIFT);
307  ret = write_register(dev, PCM512x_DIGITAL_MUTE_2, (PCM512x_VEDF | PCM512x_VEDS), val);
308  if(ret != PCM514x_ERR_OK) {
309  debug(1, "Failed to write register 0x%02x to 0x%02x: %d", PCM512x_DIGITAL_MUTE_2, val, ret);
310  return ret;
311  }
312  return PCM514x_ERR_OK;
313 }

315 pcm514x_error_t pcm514x_get_audio_ramp(pcm514x_t* dev, const pcm514x_audio_ramp_t* ramp)
316 {
317  uint8_t val;
318 
319  // get the normal mode ramps
320  read_register(dev, PCM512x_DIGITAL_MUTE_1, (PCM512x_VNDF | PCM512x_VNDS | PCM512x_VNUF | PCM512x_VNUS), &val);

324  ramp->vol_ramp_down.freq = (val & PCM512x_VNDF) >> PCM512x_VNDF_SHIFT;
325 
326  // get the emergency ramps
327  read_register(dev, PCM512x_DIGITAL_MUTE_2, (PCM512x_VEDF | PCM512x_VEDS), &val);
328  ramp->emergency_down.step = (val & PCM512x_VEDS) >> PCM512x_VEDS_SHIFT;
329  ramp->emergency_down.freq = (val & PCM512x_VEDF) >> PCM512x_VEDF_SHIFT;
330 
331  return PCM514x_ERR_OK;

334 pcm514x_error_t pcm514x_set_audio_amute(pcm514x_t* dev, const pcm514x_audio_amute_t* amute_r, const pcm514x_audio_amute_t* amute_l)
335 {
336  uint8_t val;
337  int ret;
338 
339  // setup auto mute
340  val = ((amute_r->en_amute) ? 1 : 0) << PCM512x_AMRE_SHIFT;
341  val |= ((amute_l->en_amute) ? 1 : 0) << PCM512x_AMLE_SHIFT;
342  val |= ((amute_r->en_amute == amute_l->en_amute) ? 1 : 0) << PCM512x_ACTL_SHIFT;
343  ret = write_register(dev, PCM512x_DIGITAL_MUTE_3, (PCM512x_ACTL | PCM512x_AMLE | PCM512x_AMRE), val);
344  if(ret != PCM514x_ERR_OK) {
345  debug(1, "Failed to write register 0x%02x to 0x%02x: %d", PCM512x_DIGITAL_MUTE_3, val, ret);
346  return ret;
347  }
348  val = (amute_r->amute << PCM512x_ATMR_SHIFT) | (amute_l->amute << PCM512x_ATML_SHIFT);
349  ret = write_register(dev, PCM512x_AUTO_MUTE, (PCM512x_ATMR | PCM512x_ATML), val);
350  if(ret != PCM514x_ERR_OK) {
351  debug(1, "Failed to write register 0x%02x to 0x%02x: %d", PCM512x_AUTO_MUTE, val, ret);
352  return ret;
353  }
354  return PCM514x_ERR_OK;
355 }

394 static int write_register(pcm514x_t* dev, int reg, uint8_t mask, uint8_t val)
395 {
396  int page;
397  uint8_t value;
398  pcm514x_priv_t *io = dev->io;
399 
400  // get the needed page and change to it
401  page = reg % PCM512x_PAGE_LEN - PCM512x_VIRT_BASE;
402  if(io->cur_page != page) {
403  io->write(dev, PCM512x_PAGE, page);
404  io->cur_page = page;
405  }
406 
407  // read the current register value
408  value = io->read(dev, reg);
409  // replace the masked parts
410  value = (val & mask) | (value & ~mask);
411 
412  // write the value to the register
413  // remove the page identification prior to the hw access
414  io->write(dev, reg & 0xFF, value);
415 
416  // write failed
417  if (value != io->read(dev, reg)){
418  debug(1, "SPI write at 0x%02x failed, got 0x%02x (expected: 0x%02x)!", reg, value, val);
419  return PCM514x_ERR_IOERROR;
420  }
421  return PCM514x_ERR_OK;
422 }

425 static int read_register(pcm514x_t* dev, int reg, uint8_t mask, uint8_t shift, uint8_t *val)
426 {
427  int page;
428  uint8_t value;
429  pcm514x_priv_t *io = dev->io;
430 
431  // get the needed page and change to it
432  page = reg % PCM512x_PAGE_LEN - PCM512x_VIRT_BASE;
433  if(io->cur_page != page) {
434  io->write(dev, PCM512x_PAGE, page);
435  io->cur_page = page;
436  }
437

440  value = (value & mask) >> shift;
