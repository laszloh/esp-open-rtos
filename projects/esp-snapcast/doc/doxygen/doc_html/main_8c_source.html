55 static inline void init_descriptors_list()
56 {
57  int i;
58  memset(dma_buffer, 0, DMA_BUFFER_SIZE * DMA_QUEUE_SIZE);
59 
60  for(i = 0; i < DMA_QUEUE_SIZE; i++) {
61  dma_block_list[i].owner = 1;
62  dma_block_list[i].eof = 1;
63  dma_block_list[i].sub_sof = 0;
64  dma_block_list[i].unused = 0;
65  dma_block_list[i].buf_ptr = dma_buffer[i];
66  dma_block_list[i].datalen = DMA_BUFFER_SIZE;
67  dma_block_list[i].blocksize = DMA_BUFFER_SIZE;
68  if(i == (DMA_QUEUE_SIZE - 1)){
69  // last element
70  dma_block_list[i].next_link_ptr = &dma_block_list[0];
71  } else {
72  dma_block_list[i].next_link_ptr = &dma_block_list[i+1];
73  }
74  }
75 
76  // The queue holding all empty dma buffers. it's one smaller than the
77  // real queue size, since one descriptor is currently being used by
78  // the DMA itself.
79  dma_queue = xQueueCreate(DMA_QUEUE_SIZE - 1, sizeof(uint8_t*));
80 }

82 // DMA interrupt handler. It is called each time a DMA block is finished processing.
83 static void dma_isr_handler(void)
84 {
85  portBASE_TYPE task_awoken = pdFALSE;
86  if (i2s_dma_is_eof_interrupt()) {
87  dma_descriptor_t *descr = i2s_dma_get_eof_descriptor();
88 
89  if (xQueueIsQueueFullFromISR(dma_queue)) {
90  // List of empty blocks is full. Sender don't send data fast enough.
91  int dummy;
92  underrun_counter++;
93  // Discard top of the queue
94  xQueueReceiveFromISR(dma_queue, &dummy, &task_awoken);
95  }
96  // Push the processed buffer to the queue so sender can refill it.
97  xQueueSendFromISR(dma_queue, (void*)(&descr->buf_ptr), &task_awoken);
98  }
99  i2s_dma_clear_interrupt();
100 
101  portEND_SWITCHING_ISR(task_awoken);
102 }

105 {

108  // Get a free block from the DMA queue. This call will suspend the task

115  int read_bytes = read(fd, curr_dma_buf, DMA_BUFFER_SIZE);
116  if (read_bytes <= 0) {
